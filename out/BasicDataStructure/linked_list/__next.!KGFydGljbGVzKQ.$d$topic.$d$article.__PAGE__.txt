int is_inserted[maxn]; // 節點是否在 linked list 裡面

// 初始化節點 x
void Init(int x, int prv, int nxt, int inserted) {
    prev_node[x] = prv;
    next_node[x] = nxt;
    is_inserted[x] = inserted;
}

// 在節點 x 之後插入 id 的節點
void Insert(int x, int id) {
    int nxt = next_node[x];
    Init(id, x, nxt, 1);
    prev_node[nxt] = id;
    next_node[x] = id;
}

    int prv = prev_node[x];
    int nxt = next_node[x];

}
